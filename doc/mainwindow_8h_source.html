21 public:
22  StrictIntValidator(int minimum, int maximum, QObject *parent = nullptr): QIntValidator(minimum, maximum, parent) {}
23  QValidator::State validate(QString &input, int &pos) const {
24  bool neg = false, ok;
25  if (input.startsWith('-')) {
26  if (bottom() >= 0) return QValidator::Invalid;
27  neg = true;
28  input.remove(0, 1);
29  }
30  if (!QRegExp("[0-9]*").exactMatch(input)) return QValidator::Invalid;
31  while (input.length() > 0 && input.startsWith('0')) { if (pos) --pos; input.remove(0, 1); }
32  if (input.isEmpty()) { input = "0"; pos = 1; }
33  if (neg) { input.prepend('-'); if (input == "-0") pos = 2; }
34  qint32 value = input.toInt(&ok);
35  return ok && value >= bottom() && value <= top() ? QValidator::Acceptable : QValidator::Invalid;
36  }
