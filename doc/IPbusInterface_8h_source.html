47  bool transceive(IPbusControlPacket &p, bool shouldResponseBeProcessed = true) { //send request, wait for response, receive it and check correctness
48  if (!isOnline) return false;
49  if (p.requestSize <= 1) {
50  qDebug()<<"Empty request"; //not a logicError anymore, just nothing to do
51  return true;
52  }
53  QMutexLocker ml(&mutex);
54  qint32 n = qint32(qsocket->write((char *)p.request, p.requestSize * wordSize));
55  if (n < 0) {
56  emit error("Socket write error: " + qsocket->errorString(), networkError);
57  return false;
58  } else if (n != p.requestSize * wordSize) {
59  emit error("Sending packet failed", networkError);
60  return false;
61  } else if (!qsocket->waitForReadyRead(timeout_ms) && !qsocket->hasPendingDatagrams()) {
62  isOnline = false;
63  emit noResponse();
64  return false;
65  }
66  n = qsocket->readDatagram((char *)p.response, qsocket->pendingDatagramSize());
67  if (n == 64 && p.response[0] == statusRequest.header) { //late status response received
68  if (!qsocket->hasPendingDatagrams() && !qsocket->waitForReadyRead(timeout_ms) && !qsocket->hasPendingDatagrams()) {
69  isOnline = false;
70  emit noResponse();
71  return false;
72  }
73  n = qsocket->readDatagram((char *)p.response, qsocket->pendingDatagramSize());
74  }
75  if (n == 0) {
76  emit error("empty response, no IPbus target on " + IPaddress, networkError);
77  return false;
78  } else if (n / wordSize > p.responseSize || p.response[0] != p.request[0] || n % wordSize > 0) {
79  emit error(QString::asprintf("incorrect response (%d bytes)", n), networkError);
80  return false;
81  } else {
82  p.responseSize = quint16(n / wordSize); //response can be shorter then expected if a transaction wasn't successful
83  bool result = shouldResponseBeProcessed ? p.processResponse() : true;
84  p.reset();
85  return result;
86  }
