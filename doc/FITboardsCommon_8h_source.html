426 inline quint32 changeNbits(quint32 base, quint8 length, quint8 shift, quint32 value) {
427  if (length == 0) return base;
428  if (length >= 32) return value;
429  if (length == 1) return value ? base | 1 << shift : base & ~(1 << shift);
430  quint32 mask = (1 << length) - 1;
431  base &= ~(mask << shift);
432  base |= (mask & value) << shift;
433  return base;
434 }
