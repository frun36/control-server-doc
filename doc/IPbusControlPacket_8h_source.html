39  void addTransaction(TransactionType type, quint32 address, quint32 *data, quint8 nWords = 1) {
40  Transaction currentTransaction;
41  request[requestSize] = TransactionHeader(type, nWords, transactionsList.size());
42  currentTransaction.requestHeader = (TransactionHeader *)(request + requestSize++);
43  request[requestSize] = address;
44  currentTransaction.address = request + requestSize++;
45  currentTransaction.responseHeader = (TransactionHeader *)(response + responseSize++);
46  switch (type) {
47  case ipread:
48  case nonIncrementingRead:
49  case configurationRead:
50  currentTransaction.data = data;
51  responseSize += nWords;
52  break;
53  case ipwrite:
54  case nonIncrementingWrite:
55  case configurationWrite:
56  currentTransaction.data = request + requestSize;
57  for (quint8 i=0; i<nWords; ++i) request[requestSize++] = data[i];
58  break;
59  case RMWbits:
60  request[requestSize++] = data[0]; //AND term
61  request[requestSize++] = data[1]; // OR term
62  currentTransaction.data = response + responseSize++;
63  break;
64  case RMWsum:
65  request[requestSize++] = *data; //addend
66  currentTransaction.data = response + responseSize++;
67  break;
68  default:
69  emit error("unknown transaction type", IPbusError);
70  }
71  if (requestSize > maxPacket || responseSize > maxPacket) {
72  emit error("packet size exceeded", IPbusError);
73  return;
74  } else transactionsList.append(currentTransaction);
75  }

93  case ipread:
94  case nonIncrementingRead:
95  case configurationRead: {

97  if (th->Words > wordsAhead) { //response too short to contain nWords values
98  if (transactionsList.at(i).data != nullptr) memcpy(transactionsList.at(i).data, (quint32 *)th + 1, wordsAhead * wordSize);
99  emit successfulRead(wordsAhead);
100  if (th->InfoCode == 0) emit error(QString::asprintf("read transaction from %08X truncated: %d/%d words received", *transactionsList.at(i).address, wordsAhead, th->Words), IPbusError);
101  return false;
102  } else {
103  if (transactionsList.at(i).data != nullptr) memcpy(transactionsList.at(i).data, (quint32 *)th + 1, th->Words * wordSize);
104  emit successfulRead(th->Words);
105  }
106  break;
107  }
108  case RMWbits:
109  case RMWsum :
110  if (th->Words != 1) {
111  emit error("wrong RMW transaction", IPbusError);
112  return false;
113  }
114  emit successfulRead(1);
115  /* fall through */ //[[fallthrough]];
116  case ipwrite:
117  case nonIncrementingWrite:
118  case configurationWrite:
119  emit successfulWrite(th->Words);
120  break;
121  default:
122  emit error("unknown transaction type", IPbusError);
123  return false;
124  }
125  if (th->InfoCode != 0) {
126  emit error(th->infoCodeString() + QString::asprintf(", address: %08X", *transactionsList.at(i).address + th->Words), IPbusError);
