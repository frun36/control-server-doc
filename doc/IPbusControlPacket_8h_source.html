108  void addTransaction(TransactionType type, quint32 address, quint32 *data, quint8 nWords = 1) {
109  Transaction currentTransaction;
110  request[requestSize] = TransactionHeader(type, nWords, transactionsList.size());
111  currentTransaction.requestHeader = (TransactionHeader *)(request + requestSize++);
112  request[requestSize] = address;
113  currentTransaction.address = request + requestSize++;
114  currentTransaction.responseHeader = (TransactionHeader *)(response + responseSize++);
115  switch (type) {
116  case ipread:
117  case nonIncrementingRead:
118  case configurationRead:
119  currentTransaction.data = data;
120  responseSize += nWords;
121  break;
122  case ipwrite:
123  case nonIncrementingWrite:
124  case configurationWrite:
125  currentTransaction.data = request + requestSize;
